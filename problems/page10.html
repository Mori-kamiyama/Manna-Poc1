四角形を書こう

図形描画アプリを作ります。手始めに、自由な位置に３×３の四角を表示させて下ださい

実行例

Input1
->x = 3
->y = 5

=>

Output1
_______________
_______________
_______________
_@@@___________
_@@@___________
_@@@___________
_______________
_______________

回答


#include 

int main() {
    int x, y;

    // ユーザーからxとyの入力を受け取る
    printf("xの値を入力してください: ");
    scanf("%d", &x);
    printf("yの値を入力してください: ");
    scanf("%d", &y);

    // y行分の「_」行を出力
    for (int i = 0; i < y; i++) {
        for (int j = 0; j < x + 3; j++) {  // 全幅分の「_」を出力
            printf("_");
        }
        printf("\n");
    }

    // 3x3の四角を描画
    for (int i = 0; i < 3; i++) {
        // x列分の「_」を出力
        for (int j = 0; j < x; j++) {
            printf("_");
        }
        // 3つの '@' を出力して改行
        printf("@@@\n");
    }

    // 残りの行の「_」を出力
    for (int i = y + 3; i < 8; i++) {  // 全体の高さは任意で8に設定
        for (int j = 0; j < x + 3; j++) {
            printf("_");
        }
        printf("\n");
    }

    return 0;
}
